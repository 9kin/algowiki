Бинарное возведение в степень¶

Бинарное (двоичное) возведение в степень — это приём, позволяющий возводить любое число в \(n\)-ую степень за \(O(log(n))\) умножений (вместо \(n\) умножений при обычном подходе).

Алгоритм¶

\[ \begin{align}\begin{aligned}a\in R\\n = 2k\\a^n = (a^{n/2}) ^2 = a^{n/2} * a^{n/2}\end{aligned}\end{align} \]
\[ \begin{align}\begin{aligned}a\in R\\n != 2k\\a^n = a^{n-1} * n\end{aligned}\end{align} \]
long long binpow(long long a, long long b) {
    if (b == 0)
        return 1;
    long long res = binpow(a, b / 2);
    if (b % 2)
        return res * res * a;
    else
        return res * res;
}

(деления на 2 заменены битовыми операциями)

long long binpow(long long a, long long b) {
    long long res = 1;
    while (b > 0) {
        if (b & 1)
            res = res * a;
        a = a * a;
        b >>= 1;
    }
    return res;
}
